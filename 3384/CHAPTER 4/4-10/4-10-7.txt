    r = 1
    c = 1
    
    For i = 0 To Doc.all.Length - 1
        If Doc.all(i).tagName = "TH" Or Doc.all(i).tagName = "TD" Then
            Cells(r, c) = Doc.all(i).innerText
            c = c + 1
        ElseIf Doc.all(i).tagName = "TR" Then
            r = r + 1
            c = 1
        End If
    Next i
